import { FileTree, TreeFile, TreeFolder } from "@barodoc/theme-docs/components/mdx/FileTree.tsx";
import { ParamField, ParamFieldGroup } from "@barodoc/theme-docs/components/mdx/ParamField.tsx";

파일 트리는 프로젝트 구조를 시각화하여 사용자가 디렉토리 레이아웃을 이해하는 데 도움이 됩니다.

## 기본 사용법

<FileTree client:load>
  <TreeFolder name="src" defaultOpen>
    <TreeFolder name="components" defaultOpen>
      <TreeFile name="Button.tsx" />
      <TreeFile name="Card.tsx" />
      <TreeFile name="index.ts" />
    </TreeFolder>
    <TreeFolder name="pages">
      <TreeFile name="index.astro" />
      <TreeFile name="about.astro" />
    </TreeFolder>
    <TreeFile name="styles.css" />
  </TreeFolder>
  <TreeFile name="package.json" />
  <TreeFile name="tsconfig.json" />
</FileTree>

## Barodoc 프로젝트 구조

<FileTree client:load>
  <TreeFolder name="my-docs" defaultOpen>
    <TreeFolder name="src" defaultOpen>
      <TreeFolder name="content" defaultOpen>
        <TreeFolder name="docs" defaultOpen>
          <TreeFolder name="en">
            <TreeFile name="introduction.mdx" />
            <TreeFile name="quickstart.mdx" />
          </TreeFolder>
          <TreeFolder name="ko">
            <TreeFile name="introduction.mdx" />
            <TreeFile name="quickstart.mdx" />
          </TreeFolder>
        </TreeFolder>
      </TreeFolder>
    </TreeFolder>
    <TreeFile name="astro.config.mjs" />
    <TreeFile name="barodoc.config.json" />
    <TreeFile name="package.json" />
  </TreeFolder>
</FileTree>

## Props

### FileTree

<ParamFieldGroup>
  <ParamField name="className" type="string">
    추가 CSS 클래스
  </ParamField>
</ParamFieldGroup>

### TreeFolder

<ParamFieldGroup>
  <ParamField name="name" type="string" required>
    표시할 폴더 이름
  </ParamField>
  <ParamField name="defaultOpen" type="boolean" default="false">
    기본적으로 폴더가 펼쳐져 있는지 여부
  </ParamField>
  <ParamField name="children" type="ReactNode">
    중첩된 TreeFile 및 TreeFolder 컴포넌트
  </ParamField>
</ParamFieldGroup>

### TreeFile

<ParamFieldGroup>
  <ParamField name="name" type="string" required>
    표시할 파일 이름
  </ParamField>
  <ParamField name="icon" type="ReactNode">
    커스텀 아이콘 컴포넌트 (기본값은 파일 아이콘)
  </ParamField>
</ParamFieldGroup>
