## Basic Usage

  { title: "Is it free?", content: "Yes, Barodoc is open source and free to use." }
]} />

## Multiple Open

Allow multiple items to be open simultaneously with `type="multiple"`:

<SimpleAccordion client:load type="multiple" items={[
  { title: "Frontend", content: "React, Vue, Svelte, and more are supported." },
  { title: "Backend", content: "Node.js, Python, Go, and other backends work seamlessly." },
  { title: "Database", content: "PostgreSQL, MySQL, MongoDB, and others are documented." }
]} />

```mdx
<SimpleAccordion client:load type="multiple" items={[
  { title: "Frontend", content: "React, Vue, Svelte..." },
  { title: "Backend", content: "Node.js, Python, Go..." },
  { title: "Database", content: "PostgreSQL, MySQL, MongoDB..." }
]} />
```

## Default Open

Set initially open items with `defaultValue`:

<SimpleAccordion client:load defaultValue="item-1" items={[
  { title: "First Item", content: "This is closed by default." },
  { title: "Second Item", content: "This is open by default." },
  { title: "Third Item", content: "This is also closed by default." }
]} />

```mdx
<SimpleAccordion client:load defaultValue="item-1" items={[
  { title: "First Item", content: "This is closed by default." },
  { title: "Second Item", content: "This is open by default." },
  { title: "Third Item", content: "This is also closed by default." }
]} />
```

## Props

<ParamFieldGroup>
  <ParamField name="items" type="AccordionItem[]" required>
    Array of accordion items (see Accordion Item below)
  </ParamField>
  <ParamField name="type" type="'single' | 'multiple'" default="'single'">
    Whether one or multiple items can be open at once
  </ParamField>
  <ParamField name="collapsible" type="boolean" default="true">
    Allow closing all items when type is "single"
  </ParamField>
  <ParamField name="defaultValue" type="string | string[]">
    Initially open item(s) - use "item-0", "item-1", etc.
  </ParamField>
  <ParamField name="client:load" type="directive" required>
    Required Astro directive for client-side interactivity
  </ParamField>
</ParamFieldGroup>

### Accordion Item

Each item in the `items` array:

<ParamFieldGroup>
  <ParamField name="title" type="string" required>
    Accordion header text
  </ParamField>
  <ParamField name="content" type="string" required>
    Content shown when accordion is expanded
  </ParamField>
  <ParamField name="icon" type="string">
    Optional emoji or icon to display
  </ParamField>
</ParamFieldGroup>
