import { FileTree, TreeFile, TreeFolder } from "@barodoc/theme-docs/components/mdx/FileTree.tsx";
import { ParamField, ParamFieldGroup } from "@barodoc/theme-docs/components/mdx/ParamField.tsx";

File trees visualize project structures, helping users understand directory layouts.

## Basic Usage

<FileTree client:load>
  <TreeFolder name="src" defaultOpen>
    <TreeFolder name="components" defaultOpen>
      <TreeFile name="Button.tsx" />
      <TreeFile name="Card.tsx" />
      <TreeFile name="index.ts" />
    </TreeFolder>
    <TreeFolder name="pages">
      <TreeFile name="index.astro" />
      <TreeFile name="about.astro" />
    </TreeFolder>
    <TreeFile name="styles.css" />
  </TreeFolder>
  <TreeFile name="package.json" />
  <TreeFile name="tsconfig.json" />
</FileTree>

## Props

### FileTree

<ParamFieldGroup>
  <ParamField name="className" type="string">
    Additional CSS classes
  </ParamField>
</ParamFieldGroup>

### TreeFolder

<ParamFieldGroup>
  <ParamField name="name" type="string" required>
    Folder name to display
  </ParamField>
  <ParamField name="defaultOpen" type="boolean" default="false">
    Whether folder is expanded by default
  </ParamField>
  <ParamField name="children" type="ReactNode">
    Nested TreeFile and TreeFolder components
  </ParamField>
</ParamFieldGroup>

### TreeFile

<ParamFieldGroup>
  <ParamField name="name" type="string" required>
    File name to display
  </ParamField>
  <ParamField name="icon" type="ReactNode">
    Custom icon component (defaults to file icon)
  </ParamField>
</ParamFieldGroup>
